## Description

Creates a new generator. The response contains the newly created generator data
object.

## Request URL

<div className={"doc-request-example"} style={{ marginBottom: '1em' }}>
  <div className={"method post"}>POST</div>
  <div className={"url"}>/wp-json/lmfwc/v2/generators</div>
</div>

## Request headers

```
Content-Type: application/json
```

## Request example (PHP)

```
$curl = curl_init();

curl_setopt_array($curl, array(
  CURLOPT_URL => "https://example.com/wp-json/lmfwc/v2/generators",
  CURLOPT_RETURNTRANSFER => true,
  CURLOPT_ENCODING => "",
  CURLOPT_MAXREDIRS => 10,
  CURLOPT_TIMEOUT => 0,
  CURLOPT_FOLLOWLOCATION => false,
  CURLOPT_HTTP_VERSION => CURL_HTTP_VERSION_1_1,
  CURLOPT_CUSTOMREQUEST => "POST",
  CURLOPT_POSTFIELDS =>"{
  "name": "Generator created by the API",
  "charset": "ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789",
  "chunks": 5,
  "chunk_length": 5,
  "times_activated_max": 2,
  "separator": "-",
  "prefix": null,
  "suffix": null,
  "expires_in": null
}",
  CURLOPT_HTTPHEADER => array(
    "Content-Type: application/json"
  ),
));

$response = curl_exec($curl);
$err = curl_error($curl);

curl_close($curl);

if ($err) {
  echo "cURL Error #:" . $err;
} else {
  echo $response;
}
```
